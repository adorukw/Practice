### 示例3：用PyTorch实现线性回归（手动计算梯度）

```python
import torch
import torch.nn as nn
import torch.optim as optim
import matplotlib.pyplot as plt
from sklearn.preprocessing import StandardScaler
import numpy as np

print("🔥 第二部分：PyTorch深度学习")
print("="*50)
print("📈 示例3：PyTorch线性回归（从零实现）")

# 设置随机种子保证可重复性
torch.manual_seed(42)
np.random.seed(42)

# 生成合成数据
np.random.seed(42)
n_samples = 100
X_np = 2 * np.random.rand(n_samples, 1)  # 生成0-2之间的随机数
y_np = 4 + 3 * X_np + np.random.randn(n_samples, 1) * 0.5  # y = 4 + 3x + 噪声

# 转换为PyTorch张量
X = torch.from_numpy(X_np).float()
y = torch.from_numpy(y_np).float()

# 划分训练集和测试集
train_size = int(0.8 * n_samples)
X_train, X_test = X[:train_size], X[train_size:]
y_train, y_test = y[:train_size], y[train_size:]

print(f"📊 数据形状: X_train: {X_train.shape}, y_train: {y_train.shape}")

# 方法1：手动实现梯度下降（理解原理）
print("\n" + "-"*50)
print("方法1：手动梯度下降（理解原理）")
print("-"*50)

# 初始化参数
w = torch.randn(1, requires_grad=True, dtype=torch.float32)  # 权重
b = torch.randn(1, requires_grad=True, dtype=torch.float32)  # 偏置
print(f"初始参数: w = {w.item():.3f}, b = {b.item():.3f}")

# 超参数
learning_rate = 0.01
n_epochs = 200
losses = []

# 手动训练循环
for epoch in range(n_epochs):
    # 前向传播
    y_pred = w * X_train + b
    
    # 计算损失（均方误差）
    loss = ((y_pred - y_train) ** 2).mean()
    losses.append(loss.item())
    
    # 反向传播（手动计算梯度）
    # 损失函数 L = 1/n * Σ(y_pred - y)^2
    # 对w的梯度: ∂L/∂w = 2/n * Σ(y_pred - y) * x
    # 对b的梯度: ∂L/∂b = 2/n * Σ(y_pred - y)
    grad_w = 2 * ((y_pred - y_train) * X_train).mean()
    grad_b = 2 * (y_pred - y_train).mean()
    
    # 更新参数（梯度下降）
    w = w - learning_rate * grad_w
    b = b - learning_rate * grad_b
    
    if (epoch+1) % 50 == 0:
        print(f'Epoch [{epoch+1}/{n_epochs}], Loss: {loss.item():.4f}')

print(f"\n训练后参数: w = {w.item():.3f}, b = {b.item():.3f}")
print(f"真实参数: w = 3.0, b = 4.0")

# 可视化
plt.figure(figsize=(12, 4))
plt.subplot(1, 2, 1)
plt.scatter(X_train.numpy(), y_train.numpy(), alpha=0.5, label='训练数据')
plt.scatter(X_test.numpy(), y_test.numpy(), alpha=0.5, label='测试数据', color='red')
x_range = torch.linspace(X.min(), X.max(), 100)
y_range = w * x_range + b
plt.plot(x_range.numpy(), y_range.detach().numpy(), 'r-', label='拟合直线', linewidth=3)
plt.xlabel('X')
plt.ylabel('y')
plt.title('线性回归拟合结果')
plt.legend()

plt.subplot(1, 2, 2)
plt.plot(losses)
plt.xlabel('Epoch')
plt.ylabel('Loss (MSE)')
plt.title('损失下降曲线')
plt.yscale('log')
plt.tight_layout()
plt.show()
```

#### 5.1 定义神经网络模型
```python
class SimpleNN(nn.Module):
    """简单神经网络模型"""
    def __init__(self, input_size=784, hidden_size=128, num_classes=10):
        super(SimpleNN, self).__init__()
        self.flatten = nn.Flatten()  # 将28x28图片展平为784
        self.fc1 = nn.Linear(input_size, hidden_size)  # 全连接层1
        self.relu = nn.ReLU()  # 激活函数
        self.fc2 = nn.Linear(hidden_size, hidden_size // 2)  # 全连接层2
        self.fc3 = nn.Linear(hidden_size // 2, num_classes)  # 输出层
        
    def forward(self, x):
        x = self.flatten(x)
        x = self.fc1(x)
        x = self.relu(x)
        x = self.fc2(x)
        x = self.relu(x)
        x = self.fc3(x)
        return x

# 实例化模型
model = SimpleNN().to(device)
print("🧠 模型结构:")
```

#### 6.1 定义CNN模型
```python
class SimpleCNN(nn.Module):
    """简单的CNN模型"""
    def __init__(self, num_classes=10):
        super(SimpleCNN, self).__init__()
        
        # 卷积层
        self.conv1 = nn.Conv2d(3, 32, kernel_size=3, padding=1)  # 输入3通道，输出32通道
        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
        self.conv3 = nn.Conv2d(64, 128, kernel_size=3, padding=1)
        
        # 池化层
        self.pool = nn.MaxPool2d(2, 2)
        
        # Dropout层防止过拟合
        self.dropout = nn.Dropout(0.5)
        
        # 全连接层
        # 计算全连接层的输入尺寸: 128 * 4 * 4
        # 卷积和池化后的尺寸变化: 32x32 -> 16x16 -> 8x8 -> 4x4
        self.fc1 = nn.Linear(128 * 4 * 4, 256)
        self.fc2 = nn.Linear(256, 128)
        self.fc3 = nn.Linear(128, num_classes)
        
    def forward(self, x):
        # 卷积块1: 3x32x32 -> 32x32x32 -> 32x16x16
        x = self.pool(F.relu(self.conv1(x)))
        
        # 卷积块2: 32x16x16 -> 64x16x16 -> 64x8x8
        x = self.pool(F.relu(self.conv2(x)))
        
        # 卷积块3: 64x8x8 -> 128x8x8 -> 128x4x4
        x = self.pool(F.relu(self.conv3(x)))
        
        # 展平
        x = x.view(-1, 128 * 4 * 4)
        
        # 全连接层
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        x = F.relu(self.fc2(x))
        x = self.dropout(x)
        x = self.fc3(x)
        
        return x

# 实例化模型
```

```python
def visualize_feature_maps(model, image, layer_names=['conv1', 'conv2', 'conv3']):
    """可视化卷积层的特征图"""
    model.eval()
    
    # 前向传播并获取中间特征
    activations = {}
    
    def get_activation(name):
        def hook(model, input, output):
            activations[name] = output.detach()
        return hook
    
    # 注册钩子
    hooks = []
    for name, layer in model.named_modules():
        if name in layer_names:
            hooks.append(layer.register_forward_hook(get_activation(name)))
    
    # 前向传播
    with torch.no_grad():
        _ = model(image.unsqueeze(0).to(device))
    
    # 移除钩子
    for hook in hooks:
        hook.remove()
    
    # 可视化特征图
    fig, axes = plt.subplots(len(layer_names), 8, figsize=(15, 3*len(layer_names)))
    if len(layer_names) == 1:
        axes = axes.reshape(1, -1)
    
    for i, (layer_name, activation) in enumerate(activations.items()):
        # 获取前8个特征图
        feature_maps = activation[0, :8].cpu()
        
        for j in range(min(8, feature_maps.size(0))):
            ax = axes[i, j] if len(layer_names) > 1 else axes[j]
            ax.imshow(feature_maps[j], cmap='viridis')
            ax.axis('off')
            if j == 0:
                ax.set_title(f'{layer_name}', fontsize=12)
    
    plt.suptitle('CNN各层特征图可视化', fontsize=16, y=1.02)
    plt.tight_layout()
    plt.show()
```
