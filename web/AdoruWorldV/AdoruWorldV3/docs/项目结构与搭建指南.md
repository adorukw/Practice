## 三、从零搭建步骤

### 第 1 步：创建前端项目

```bash
# 用 Vite 创建 Vue 3 + TypeScript 项目
npm create vite@latest AdoruWorld -- --template vue-ts

cd AdoruWorld
```

### 第 2 步：安装前端依赖

```bash
# 核心依赖
npm install vue-router pinia marked highlight.js

# Tailwind CSS v4（注意 v4 的安装方式与 v3 不同）
npm install tailwindcss @tailwindcss/vite

# 开发依赖
npm install -D @types/node
```

### 第 3 步：配置 Vite

创建 `vite.config.ts`：

```typescript
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import tailwindcss from '@tailwindcss/vite'
import { fileURLToPath, URL } from 'node:url'

export default defineConfig({
  plugins: [vue(), tailwindcss()],
  resolve: {
    alias: {
      '@': fileURLToPath(new URL('./src', import.meta.url))
    }
  },
  server: {
    proxy: {
      '/api': {
        target: 'http://localhost:3002',
        changeOrigin: true
      },
      '/uploads': {
        target: 'http://localhost:3002',
        changeOrigin: true
      }
    }
  }
})
```

### 第 4 步：配置 TypeScript 路径别名

在 `tsconfig.app.json` 中添加：

```json
{
  "compilerOptions": {
    "baseUrl": ".",
    "paths": {
      "@/*": ["./src/*"]
    }
  }
}
```

### 第 5 步：配置 Tailwind CSS

在 `src/style.css` 顶部引入 Tailwind：

```css
@import "tailwindcss";

@theme {

```

### 第 6 步：配置入口 HTML

在 `index.html` 的 `<head>` 中引入像素字体：

```html
<link rel="preconnect" href="https://fonts.googleapis.com">
<link rel="preconnect" href="https://fonts.gstatic.com" crossorigin>
<link href="https://fonts.googleapis.com/css2?family=Noto+Sans+SC:wght@400;500;700&display=swap" rel="stylesheet">
```

像素字体 `Press Start 2P` 在 `style.css` 中通过 `@font-face` 引入。

### 第 7 步：创建后端项目

```bash
# 在项目根目录下创建 server 文件夹
mkdir server
cd server

# 初始化 package.json
npm init -y
```

修改 `server/package.json` 中的关键字段：

```json
{
  "type": "commonjs",
  "scripts": {
    "dev": "tsx watch src/index.ts",
    "build": "tsc",
    "start": "node dist/index.js",
    "db:migrate": "npx prisma migrate dev",
    "db:seed": "tsx prisma/seed.ts",
    "db:studio": "npx prisma studio",
    "db:generate": "npx prisma generate"
  }
}
```

### 第 8 步：安装后端依赖

```bash
cd server

# 核心依赖
npm install express @prisma/client prisma cors multer marked uuid

# 开发依赖
npm install -D typescript tsx @types/express @types/cors @types/multer @types/uuid
```

### 第 9 步：初始化 Prisma + SQLite

```bash
cd server

# 初始化 Prisma（使用 SQLite）
npx prisma init --datasource-provider sqlite
```

这会创建 `prisma/schema.prisma` 文件，然后编写数据模型：

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = "file:./dev.db"
}

model Post {
  id          String   @id @default(uuid())
  slug        String   @unique
  title       String
  description String   @default("")
  content     String   @default("")
  coverImage  String?
  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")
  published   Boolean  @default(false)
  wordCount   Int      @default(0) @map("word_count")
  views       Int      @default(0)
  featured    Boolean  @default(false)
  categories  PostCategory[]
  tags        PostTag[]
  @@map("posts")
}

model Category {
  id          String   @id @default(uuid())
  name        String   @unique
  slug        String   @unique
  description String?
  icon        String?
  color       String?
  posts       PostCategory[]
  dexEntries  DexEntry[]
  @@map("categories")
}

model Tag {
  id    String @id @default(uuid())
  name  String @unique
  slug  String @unique
  color String?
  posts PostTag[]
  @@map("tags")
}

model PostCategory {
  postId     String   @map("post_id")
  categoryId String   @map("category_id")
  post       Post     @relation(fields: [postId], references: [id], onDelete: Cascade)
  category   Category @relation(fields: [categoryId], references: [id], onDelete: Cascade)
  @@id([postId, categoryId])
  @@map("post_categories")
}

model PostTag {
  postId String @map("post_id")
  tagId  String @map("tag_id")
  post   Post   @relation(fields: [postId], references: [id], onDelete: Cascade)
  tag    Tag    @relation(fields: [tagId], references: [id], onDelete: Cascade)
  @@id([postId, tagId])
  @@map("post_tags")
}

model Project {
  id          String @id @default(uuid())
  name        String
  description String @default("")
  tech        String @default("[]")
  link        String?
  github      String?
  image       String?
  status      String @default("in-progress")
  @@map("projects")
}

model DexEntry {
  id            String   @id @default(uuid())
  slug          String   @unique
  title         String
  originalTitle String?  @map("original_title")
  cover         String
  category      String
  status        String
  rating        Int      @default(0)
  progress      Int      @default(0)
  total         Int      @default(1)
  startDate     String?  @map("start_date")
  finishDate    String?  @map("finish_date")
  tags          String   @default("[]")
  comment       String?
  creator       String?
  year          Int?
  platform      String?
  genre         String   @default("[]")
  createdAt     DateTime @default(now()) @map("created_at")
  updatedAt     DateTime @updatedAt @map("updated_at")
  categoryId    String?  @map("category_id")
  categoryRef   Category? @relation(fields: [categoryId], references: [id], onDelete: SetNull)
  @@map("dex_entries")
}

model SiteConfig {
  id    String @id @default("site")
  key   String @unique
  value String @default("{}")
  @@map("site_configs")
}
```

### 第 10 步：运行数据库迁移 + 种子数据

```bash
cd server

# 生成 Prisma Client
npx prisma generate

# 创建数据库并运行迁移
npx prisma migrate dev --name init

# 插入种子数据
tsx prisma/seed.ts
```

### 第 11 步：创建后端入口文件

创建 `server/src/db.ts`：

```typescript
import { PrismaClient } from '@prisma/client'
const prisma = new PrismaClient()
export default prisma
```

创建 `server/src/index.ts`：

```typescript
import express from 'express'
import cors from 'cors'
import path from 'path'
import postsRouter from './routes/posts'
import categoriesRouter from './routes/categories'
import tagsRouter from './routes/tags'
import projectsRouter from './routes/projects'
import dexRouter from './routes/dex'
import siteRouter from './routes/site'
import uploadRouter from './routes/upload'

const app = express()
const PORT = process.env.PORT || 3002

app.use(cors())
app.use(express.json({ limit: '10mb' }))
app.use('/uploads', express.static(path.resolve(__dirname, '../uploads')))

app.use('/api/posts', postsRouter)
app.use('/api/categories', categoriesRouter)
app.use('/api/tags', tagsRouter)
app.use('/api/projects', projectsRouter)
app.use('/api/dex', dexRouter)
app.use('/api/site', siteRouter)
app.use('/api/upload', uploadRouter)

app.get('/api/health', (_req, res) => {
  res.json({ status: 'ok' })
})

app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`)
})
```

创建 `server/tsconfig.json`：

```json
{
  "compilerOptions": {
    "target": "ES2020",
    "module": "commonjs",
    "lib": ["ES2020"],
    "outDir": "./dist",
    "rootDir": "./src",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    "resolveJsonModule": true,
    "declaration": true,
    "sourceMap": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules", "dist"]
}
```

## 八、数据库模型关系

```
Post ←→ PostCategory ←→ Category
Post ←→ PostTag ←→ Tag
Category ←→ DexEntry（一对多，可选关联）
Project（独立表）
SiteConfig（独立表，存储 JSON 配置）
```

关键设计：
- Post 与 Category、Tag 通过中间表实现多对多关系
- DexEntry 的 category 字段是字符串（用于筛选），categoryId 是外键（关联 Category 表）
- Project 的 tech 字段存储 JSON 数组字符串
- SiteConfig 用 key-value 模式存储灵活配置（如作者信息）

### 核心组件样式

所有组件统一使用**像素双线边框**设计语言：

```css
/* 像素双线边框：内边框 → 白色间隙 → 外边框 */
box-shadow:
  0 0 0 2px var(--color-pokemon-white),   /* 白色间隙 */
  0 0 0 4px var(--color-pokemon-black),   /* 外边框 */
  4px 4px 0 0 var(--color-pokemon-dark-gray),  /* 投影层1 */
  8px 8px 0 0 var(--color-pokemon-black);      /* 投影层2 */
```

### 字体

- **像素字体**：`Press Start 2P`，用于标题、按钮、标签
- **正文字体**：`Noto Sans SC`，用于文章内容、描述

### 动画

- **云朵飘动**：Header 中 5 朵云，不同速度和延迟
- **青草摇曳**：Footer 中 `grass-sway` 动画，steps() 实现像素化
- **悬浮反馈**：卡片/按钮 hover 时 translate(-2px, -2px) + 阴影加深
- **经验值条**：`steps(12)` 实现像素化填充动画

## 十、打包部署

```bash
# 前端构建
npm run build
# 产物在 dist/ 目录

# 后端构建
cd server && npm run build
# 产物在 server/dist/ 目录

# 生产环境启动后端
cd server && npm start
```

部署时需要：
1. 用 Nginx 托管前端 `dist/` 目录
2. Nginx 反向代理 `/api` 和 `/uploads` 到后端 3002 端口
3. 后端进程用 pm2 管理：`pm2 start server/dist/index.js`
